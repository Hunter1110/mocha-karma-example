(function() {
    
    /**
     * 默认的一些键值对
     */
    var defaultStore = Config.webEjsEnvStore;
    
    defaultStore && Util.each(defaultStore, function(key, value) {
        key && localStorage.setItem(key, localStorage.getItem(key) || value);
    });
 
})();
